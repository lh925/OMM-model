;GLUH=THETA(4)*EXP(ETA(4))

IF (TIME.GE.30.AND.TIME.LE.60) LGLU=GLU+GSLOPE*(TIME-30)

IF (TIME.GE.60.AND.TIME.LE.90) LGLU=GLU+GSLOPE*(TIME-60)

IF (TIME.GE.90.AND.TIME.LE.120) LGLU=GLU+GSLOPE*(TIME-90)

IF (TIME.GE.120.AND.TIME.LE.150) LGLU=GLU+GSLOPE*(TIME-120)

IF (TIME.GE.150.AND.TIME.LT.180) LGLU=GLU+GSLOPE*(TIME-150)

IF (TIME.GE.180.AND.TIME.LT.240) LGLU=GLU+GSLOPE*(TIME-180)

IF (TIME.EQ.240) LGLU=GLU

 

IF (TIME.EQ.0) LDER=DERG

IF (TIME.GT.0.AND.TIME.LE.30) LDER=DERG+SLOPE*TIME

IF (TIME.GE.30.AND.TIME.LE.60) LDER=DERG+SLOPE*(TIME-30)

IF (TIME.GE.60.AND.TIME.LE.90) LDER=DERG+SLOPE*(TIME-60)

IF (TIME.GE.90.AND.TIME.LE.120) LDER=DERG+SLOPE*(TIME-90)

IF (TIME.GE.120.AND.TIME.LE.150) LDER=DERG+SLOPE*(TIME-120)

IF (TIME.GE.150.AND.TIME.LT.180) LDER=DERG+SLOPE*(TIME-150)

IF (TIME.GE.180.AND.TIME.LT.240) LDER=DERG+SLOPE*(TIME-180)

IF (TIME.EQ.240) LDER=DERG

CONS=((LGLU-GSS)+ABS(LGLU-GSS))/(2*(LGLU-GSS))

LLDER=(LDER+ABS(LDER))/2*CONS

PHIS=BETA*18.016

PHIB=K01*CPB/GSS*18.016

PHITOT=KD*DELTAG *18.016/(GAUC-GLUH*180)+PHIS

Q3EF=((LGLU-GLUH)*BETA+SRB+ABS((LGLU-GLUH)*BETA+SRb))/2

;;;;;;;;;;;;;;;;;Initial Value ;;;;;;;;;;;;;;;;;;

;A_0FLG is set by NONMEM for the first record of each subject (i.e. NEWIND.LE.0)

ENDIF

DADT(4)=-(K01+K21)*A(4)+K12*A(2)+A(3)+KD*LLDER

0.04

$SUBROUTINE ADVAN13 TOL=6

$MODEL      NCOMPARTMENTS=2 COMP=(CGLU,DEFOBS) ;glucose compartment

COMP=INSE ; insulin effect

$PK

TVFVG=THETA(1)

FVG=THETA(1)*EXP(ETA(1))

TVSG=THETA(2)

SG=THETA(2)*EXP(ETA(2))

P2=DEXP(THETA(3)+ETA(3))

TVP3=THETA(4)

P3=THETA(4)*EXP(ETA(4))/1000000

TVTabs=THETA(5)

Tabs=THETA(5)*EXP(ETA(5))

TVkel=THETA(6)

kel=THETA(6)*EXP(ETA(6))

SI=P3/P2*FVG*10000 ;

DOSE1=75000

IF (TIME.LT.Tabs) THEN

RA=DOSE1*FVG*kel/Tabs/WT/kel*(1-EXP(-kel*TIME))

ELSE

RA=DOSE1*FVG*kel/Tabs/WT/kel*(1-EXP(-kel*Tabs))*EXP(-kel*(TIME-Tabs))

ENDIF

IF (A_0FLG.EQ.1) THEN

A_0(1)=GB

A_0(2)   =0

ENDIF

$DES

DADT(1)=SG*GB-(SG+A(2))*A(1)+RA

DADT(2)=-P2*A(2)+P3*(INS-IB)

$ERROR

CGLU=A(1)

IPRED=CGLU

Y=IPRED+ERR(1)*IPRED+ERR(2)

$THETA

(0, 0.776,1.5) ; FVG dL/kg

(0, 0.0072,0.25) ; SG

(-4.41) FIX; P2

(0, 4.6,100); P3

(0, 144,200); Tabs

(0, 0.0532,10); kel

 0.392;         SG

 0.0436;         P3 10

 0 FIX;         ka

 0.547;          kel 2

 0.00435
